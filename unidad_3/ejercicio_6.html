Trabajo Autónomo No. 06

Objetivos

Desarrollar una aplicación en C# para la gestión de calificaciones de un grupo de alumnos utilizando los principios de la Programación Orientada a Objetos. Mediante esta actividad se busca comprender la creación de clases, propiedades y métodos, así como la implementación de operaciones de gestión de datos y el cálculo automático de calificaciones a partir de la nota obtenida por cada estudiante. Además, esta actividad fortalece el pensamiento crítico y reflexivo en la resolución de problemas.

Descripción del Problema

Se requiere desarrollar un sistema que permita administrar la información académica de un grupo de alumnos.

Cada alumno estará caracterizado por los siguientes datos:

  • DNI
  • Apellidos
  • Nombre
  • Nota
  • Calificación

La calificación será calculada automáticamente según la nota obtenida:

  • SS (Suspenso): Nota menor que 5.
  • AP (Aprobado): Nota entre 5 y menor que 7.
  • NT (Notable): Nota entre 7 y menor que 9.
  • SB (Sobresaliente): Nota mayor o igual a 9.

Funciones del Sistema

  • Mostrar alumnos registrados.
  • Introducir un nuevo alumno.
  • Eliminar alumno por DNI.
  • Consultar nota y calificación por DNI.
  • Modificar nota de un alumno.
  • Mostrar alumnos suspensos.
  • Mostrar alumnos aprobados.
  • Mostrar candidatos a Matrícula de Honor.
  • Modificar calificación manualmente.

Diseño de la Clase Alumno

class Alumno
{
    public string DNI { get; set; }
    public string Apellidos { get; set; }
    public string Nombre { get; set; }
    public double Nota { get; set; }
    public string Calificacion { get; set; }
}

Cálculo Automático de la Calificación

public static string ObtenerCalificacion(
    double nota)
{
    if (nota < 5)
        return "SS";

    if (nota < 7)
        return "AP";

    if (nota < 9)
        return "NT";

    return "SB";
}

Código en C#

using System;
using System.Collections.Generic;

class Alumno
{
    public string DNI { get; set; }
    public string Apellidos { get; set; }
    public string Nombre { get; set; }
    public double Nota { get; set; }
    public string Calificacion { get; set; }

    public void ActualizarCalificacion()
    {
        if (Nota < 5)
            Calificacion = "SS";
        else if (Nota < 7)
            Calificacion = "AP";
        else if (Nota < 9)
            Calificacion = "NT";
        else
            Calificacion = "SB";
    }
}

class Program
{
    static List alumnos =
        new List();

    static void Main()
    {
        int opcion;

        do
        {
            Console.WriteLine("\n=== GESTIÓN DE ALUMNOS ===");
            Console.WriteLine("1. Introducir alumno");
            Console.WriteLine("2. Mostrar alumnos");
            Console.WriteLine("3. Consultar alumno");
            Console.WriteLine("4. Modificar nota");
            Console.WriteLine("5. Eliminar alumno");
            Console.WriteLine("6. Mostrar suspensos");
            Console.WriteLine("7. Mostrar aprobados");
            Console.WriteLine("8. Mostrar candidatos a MH");
            Console.WriteLine("9. Salir");

            opcion = int.Parse(Console.ReadLine());

            switch (opcion)
            {
                case 1:
                    IntroducirAlumno();
                    break;
                case 2:
                    MostrarAlumnos();
                    break;
                case 3:
                    ConsultarAlumno();
                    break;
                case 4:
                    ModificarNota();
                    break;
                case 5:
                    EliminarAlumno();
                    break;
                case 6:
                    MostrarSuspensos();
                    break;
                case 7:
                    MostrarAprobados();
                    break;
                case 8:
                    MostrarMH();
                    break;
            }

        } while (opcion != 9);
    }

    static void IntroducirAlumno()
    {
        Alumno alumno = new Alumno();

        Console.Write("DNI: ");
        alumno.DNI = Console.ReadLine();

        foreach (Alumno a in alumnos)
        {
            if (a.DNI == alumno.DNI)
            {
                Console.WriteLine(
                    "Ya existe un alumno con ese DNI."
                );
                return;
            }
        }

        Console.Write("Apellidos: ");
        alumno.Apellidos = Console.ReadLine();

        Console.Write("Nombre: ");
        alumno.Nombre = Console.ReadLine();

        Console.Write("Nota: ");
        alumno.Nota =
            double.Parse(Console.ReadLine());

        alumno.ActualizarCalificacion();

        alumnos.Add(alumno);

        Console.WriteLine(
            "Alumno registrado correctamente."
        );
    }

    static void MostrarAlumnos()
    {
        foreach (Alumno a in alumnos)
        {
            Console.WriteLine(
                a.DNI + " | " +
                a.Apellidos + " | " +
                a.Nombre + " | " +
                a.Nota + " | " +
                a.Calificacion
            );
        }
    }

    static void ConsultarAlumno()
    {
        Console.Write("Ingrese DNI: ");
        string dni = Console.ReadLine();

        foreach (Alumno a in alumnos)
        {
            if (a.DNI == dni)
            {
                Console.WriteLine(
                    "Nombre: " + a.Nombre
                );
                Console.WriteLine(
                    "Nota: " + a.Nota
                );
                Console.WriteLine(
                    "Calificación: " +
                    a.Calificacion
                );
                return;
            }
        }

        Console.WriteLine("Alumno no encontrado.");
    }

    static void ModificarNota()
    {
        Console.Write("Ingrese DNI: ");
        string dni = Console.ReadLine();

        foreach (Alumno a in alumnos)
        {
            if (a.DNI == dni)
            {
                Console.Write("Nueva nota: ");
                a.Nota =
                    double.Parse(Console.ReadLine());

                a.ActualizarCalificacion();

                Console.WriteLine(
                    "Nota actualizada."
                );
                return;
            }
        }

        Console.WriteLine("Alumno no encontrado.");
    }

    static void EliminarAlumno()
    {
        Console.Write("Ingrese DNI: ");
        string dni = Console.ReadLine();

        for (int i = 0; i < alumnos.Count; i++)
        {
            if (alumnos[i].DNI == dni)
            {
                alumnos.RemoveAt(i);

                Console.WriteLine(
                    "Alumno eliminado."
                );
                return;
            }
        }

        Console.WriteLine("Alumno no encontrado.");
    }

    static void MostrarSuspensos()
    {
        foreach (Alumno a in alumnos)
        {
            if (a.Nota < 5)
            {
                Console.WriteLine(
                    a.Nombre + " - " +
                    a.Nota
                );
            }
        }
    }

    static void MostrarAprobados()
    {
        foreach (Alumno a in alumnos)
        {
            if (a.Nota >= 5)
            {
                Console.WriteLine(
                    a.Nombre + " - " +
                    a.Nota
                );
            }
        }
    }

    static void MostrarMH()
    {
        foreach (Alumno a in alumnos)
        {
            if (a.Nota == 10)
            {
                Console.WriteLine(
                    a.Nombre + " - MH"
                );
            }
        }
    }
}
